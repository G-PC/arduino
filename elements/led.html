💡 Подключение: анод (+) через резистор 220 Ом к пину 9, катод (-) к GND.

💻 Код для Arduino IDE

// Мигание светодиодом на пине 9
int ledPin = 9;

void setup() {
  pinMode(ledPin, OUTPUT);
}

void loop() {
  digitalWrite(ledPin, HIGH);
  delay(500);
  digitalWrite(ledPin, LOW);
  delay(500);
}
