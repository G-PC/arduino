/*
 * Светодиод (LED) + Arduino Uno
 * 
 * Подключение:
 *   Анод (+) светодиода → резистор 220 Ом → цифровой пин 9
 *   Катод (-) светодиода → GND (земля)
 * 
 * Описание работы:
 *   Светодиод мигает с интервалом 500 миллисекунд (0.5 секунды)
 */

// ========== НАСТРОЙКИ ==========
// Определяем пин, к которому подключён светодиод
int ledPin = 9;   // Пин 9 поддерживает ШИМ (~) для плавной регулировки яркости

// ========== БЛОК setup() ==========
// Выполняется один раз при включении или перезагрузке Arduino
void setup() {
  // Настраиваем пин ledPin как ВЫХОД (OUTPUT)
  // OUTPUT означает, что Arduino будет подавать напряжение на этот пин
  pinMode(ledPin, OUTPUT);
}

// ========== БЛОК loop() ==========
// Выполняется бесконечно снова и снова после setup()
void loop() {
  // ВКЛЮЧАЕМ светодиод
  // digitalWrite(pin, HIGH) подаёт 5V на указанный пин
  digitalWrite(ledPin, HIGH);
  
  // ЖДЁМ 500 миллисекунд (0.5 секунды)
  // Светодиод горит всё это время
  delay(500);
  
  // ВЫКЛЮЧАЕМ светодиод
  // digitalWrite(pin, LOW) подаёт 0V (землю) на указанный пин
  digitalWrite(ledPin, LOW);
  
  // ЖДЁМ 500 миллисекунд (0.5 секунды)
  // Светодиод не горит всё это время
  delay(500);
  
  // Цикл повторяется: загорелся → потух → загорелся → потух...
}
